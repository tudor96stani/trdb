9    /// Internal method for planning the insert of a record in a heap unsorted page
10    pub(super) fn plan_insert_internal(
11        &self,
12        row_len: usize,
13    ) -> Result<InsertionPlan, InsertError> {
14        let header = self.header_ref()?;
15        let slot_array_ref = self.slot_array_ref()?;
16        let free_start = header.get_free_start()? as usize;
17        let free_end = header.get_free_end()? as usize;
18        let slot_count = header.get_slot_count()? as usize;
19
20        if !self.row_size_fits(row_len)? {
21            return Err(InsertError::NotEnoughSpace {
22                row_len,
23                page_free_space: header.get_free_space()? as usize,
24            });
25        }
26
27        // Attempt to look for an empty slot that we can reuse from a row that has been deleted.
28        let mut unused_slot: Option<usize> = None;
29
30        // While going through the slot array, we will also look for available space,
31        // in case the row cannot fit between free start & end.
32        let mut available_space_start = if free_end - free_start > row_len {
33            Some(free_start)
34        } else {
35            None
36        };
37
38        // Just a flag that will be used down the line
39        // -> namely, we need to know whether we are adding the new row at the end of the data section (basically extending the data section)
40        // or if we have found some fragment of space in between existing rows that we can leverage
41        // For starters -> we can check if we are from the start using the freeStart as the start offset of the row.
42        let mut inserting_at_free_start = available_space_start.is_some();
43
44        for slot_index in 0..slot_count {
45            // If we found both things we were looking for, we can terminate early.
46            if unused_slot.is_some() && available_space_start.is_some() {
47                break;
48            }
49
50            if unused_slot.is_none() {
51                // Let's check if this current slot position is empty and can be used
52                let current_slot = slot_array_ref.slot_ref(slot_index as u32)?;
53                if !self.is_slot_valid(&current_slot)? {
54                    unused_slot = Some(slot_index)
55                }
56            }
57
58            // Only go up to the next-to-last slot for this check.
59            // example: inserting 10 bytes, with this slot array:
60            //        current,      next
61            // ....., (1050, 50), (1120, 30),....
62            // which means the data in the page looks like ( . = used space, _ = free space)
63            //     1050      1099  1100       1119   1120
64            // .... a ........ a    _ ___________     b .......
65            // if 1120 - 1050 - 50 = 20 > 10 bytes we want to insert => we can use this space
66            // the start position will be at 1050 + 50 = 1100
67            if available_space_start.is_none() && slot_index < slot_count - 1 {
68                let current_slot = slot_array_ref.slot_ref(slot_index as u32)?;
69                let next_slot = slot_array_ref.slot_ref((slot_index + 1) as u32)?;
70                let gap = next_slot
71                    .offset()?
72                    .checked_sub(current_slot.offset()? + current_slot.length()?);
73                if let Some(g) = gap
74                    && g >= row_len as u16
75                    && self.is_slot_valid(&next_slot)?
76                    && self.is_slot_valid(&current_slot)?
77                {
78                    available_space_start =
79                        Some((current_slot.offset()? + current_slot.length()?) as usize);
80                }
81            }
82        }
83
84        // No unused slot found in the slot array -> we will append a new one at the end of the array
85        if unused_slot.is_none() {
86            unused_slot = Some(slot_count);
87        }
88
89        let can_compact = header.get_can_compact()?;
90
91        // We could not find any available space. We should try to compact
92        let needs_compaction =
93            available_space_start.is_none() && header.needs_compaction(row_len)?;
94
95        if needs_compaction {
96            // Since we have already compacted, we will extend the data section.
97            // freeStart now holds the next available position
98            // todo: this is fully not correct. in theory, yes, the free_start is the correct value at which we will be inserting, but it's not the **current** free_start, but rather the one that will be computed after the compaction. as long as the insert algorithm handles this and adjusts the offset at which it does the insert, it is fine though.
99            available_space_start = Some(free_start);
100
101            // Also, set this flag to true, regardless if the value it had had.
102            inserting_at_free_start = true;
103        }
104
105        // todo should this ever happen??
106        if available_space_start.is_none() {
107            return Err(InsertError::CannotFindSpace {
108                required_space: row_len,
109            });
110        }
111
112        // Should be safe to unwrap here. Above, we assigned a value to unused_slot if it was None
113        let unused_slot = unused_slot.unwrap();
114        let inserting_new_slot = unused_slot == slot_count;
115
116        Ok(InsertionPlan {
117            slot_number: unused_slot as u16,
118            start_offset: available_space_start.unwrap(),
119            inserting_at_free_start,
120            inserting_new_slot,
121            needs_compaction,
122        })
123    }

125    pub(super) fn insert_row_unsorted_internal(
126        &mut self,
127        plan: InsertionPlan,
128        bytes: Vec<u8>,
129    ) -> Result<(), InsertError> {
130        let mut header_mut = self.header_mut()?;
131
132        // Could not find available space during the insertion planning => compact now if needed
133        if plan.needs_compaction {
134            todo!("Compaction not yet implemented")
135        }
136
137        // If we are inserting at free start, use the *current* freeStart after potential compaction.
138        // Otherwise, use the offset we planned in the gap.
139        let start_offset = if plan.inserting_at_free_start {
140            header_mut.get_free_start()? as usize
141        } else {
142            plan.start_offset
143        };
144
145        // Update the total slot count in the header - only increment it if we are generating a new slot
146        let updated_total_slots =
147            header_mut.get_slot_count()? + if plan.inserting_new_slot { 1 } else { 0 };
148        header_mut.set_slot_count(updated_total_slots)?;
149
150        // Only update free start if we are inserting there - if we are placing it in some free fragment, free start can stay the same
151        if plan.inserting_at_free_start {
152            let new_free_start = header_mut.get_free_start()? + bytes.len() as u16;
153            header_mut.set_free_start(new_free_start)?
154        }
155
156        // Only update the free end if we are creating a new slot - if we are re-using one of the existing unused ones, the free end can stay the same
157        if plan.inserting_new_slot {
158            let new_free_end = header_mut.get_free_end()? - SLOT_SIZE as u16;
159            header_mut.set_free_end(new_free_end)?
160        }
161
162        // freeSpace is always updated -> but depending on whether we are creating a new slot or reusing an existing one,
163        // we need to subtract the size of a slot (for creating a new slot) or 0 (for reusing an existing slot)
164        let new_free_space = header_mut.get_free_space()?
165            - bytes.len() as u16
166            - if plan.inserting_new_slot {
167                SLOT_SIZE as u16
168            } else {
169                0
170            };
171        header_mut.set_free_space(new_free_space)?;
172
173        // Insert the actual row in the page
174        self.data[start_offset..(start_offset + bytes.len())].copy_from_slice(bytes.as_slice());
175
176        // Right now, the header should contain the latest information regarding the number of slots (whether we added a new one or not),
177        // so the slot array should include any new bytes segment. We place the new slot directly at the requested index
178        self.slot_array_mut()?.set_slot(
179            plan.slot_number as u32,
180            start_offset as u16,
181            bytes.len() as u16,
182        );
183
184        Ok(())
185    }

187    fn is_slot_valid(&self, slot: &SlotRef) -> Result<bool, SlotError> {
188        Ok(slot.length()? != 0 && slot.offset()? != 0)
