9    pub(super) fn plan_insert_internal(
10        &self,
11        row_len: usize,
12    ) -> Result<InsertionPlan, InsertError> {
13        let header = self.header_ref();
14
15        if !self.row_size_fits(row_len)? {
16            return Err(InsertError::NotEnoughSpace {
17                row_len,
18                page_free_space: header.get_free_space()? as usize,
19            });
20        }
21
22        let mut unused_slot: Option<u32> = None;
23        let mut available_space_start =
24            if (header.get_free_end()? - header.get_free_start()?) as usize > row_len {
25                Some(header.get_free_start()? as usize)
26            } else {
27                None
28            };
29
30        let mut inserting_at_free_start = available_space_start.is_some();
31
32        for slot_index in 0..header.get_slot_count()? {
33            if unused_slot.is_some() && available_space_start.is_some() {
34                break;
35            }
36
37            if unused_slot.is_none() {
38                let current_slot = self.slot_array()?.slot_ref(slot_index as u32)?;
39                if current_slot.length()? == 0 && current_slot.offset()? == 0 {
40                    unused_slot = Some(slot_index as u32)
41                }
42            }
43
44            if available_space_start.is_none() && slot_index < header.get_slot_count()? - 1 {
45                let current_slot = self.slot_array()?.slot_ref(slot_index as u32)?;
46                let next_slot = self.slot_array()?.slot_ref((slot_index + 1) as u32)?;
47                let gap = next_slot.offset()? - (current_slot.offset()? + current_slot.length()?);
48                if gap >= row_len as u16 {
49                    available_space_start =
50                        Some((current_slot.offset()? + current_slot.length()?) as usize);
51                }
52            }
53        }
54
55        if unused_slot.is_none() {
56            unused_slot = Some(header.get_slot_count()? as u32);
57        }
58
59        let needs_compaction = available_space_start.is_none()
60            && header.get_can_compact()? != 0
61            && header.needs_compaction(row_len)?;
62
63        if needs_compaction {
64            available_space_start = Some(header.get_free_start()? as usize);
65            inserting_at_free_start = true;
66        }
67
68        if available_space_start.is_none() {
69            return Err(InsertError::CannotFindSpace {
70                required_space: row_len,
71            });
72        }
73
74        let inserting_new_slot = unused_slot.unwrap() == header.get_slot_count()? as u32;
75
76        Ok(InsertionPlan {
77            slot_number: unused_slot.unwrap() as u16,
78            start_offset: available_space_start.unwrap(),
79            inserting_at_free_start,
80            inserting_new_slot,
81            needs_compaction,
82        })
83    }

85    pub(super) fn insert_row_unsorted_internal(
86        &mut self,
87        plan: InsertionPlan,
88        bytes: Vec<u8>,
89    ) -> Result<(), InsertError> {
90        let mut header_mut = self.header_mut();
91
92        // Could not find available space during the insertion planning => compact now if needed
93        if plan.needs_compaction {
94            todo!("Compaction not yet implemented")
95        }
96
97        // If we are inserting at free start, use the *current* freeStart after potential compaction.
98        // Otherwise, use the offset we planned in the gap.
99        let start_offset = if plan.inserting_at_free_start {
100            header_mut.get_free_start()? as usize
101        } else {
102            plan.start_offset
103        };
104
105        // Update the total slot count in the header - only increment it if we are generating a new slot
106        let updated_total_slots =
107            header_mut.get_slot_count()? + if plan.inserting_new_slot { 1 } else { 0 };
108        header_mut.set_slot_count(updated_total_slots)?;
109
110        // Only update free start if we are inserting there - if we are placing it in some free fragment, free start can stay the same
111        if plan.inserting_at_free_start {
112            let new_free_start = header_mut.get_free_start()? + bytes.len() as u16;
113            header_mut.set_free_start(new_free_start)?
114        }
115
116        // Only update the free end if we are creating a new slot - if we are re-using one of the existing unused ones, the free end can stay the same
117        if plan.inserting_new_slot {
118            let new_free_end = header_mut.get_free_end()? - SLOT_SIZE as u16;
119            header_mut.set_free_end(new_free_end)?
120        }
121
122        // freeSpace is always updated -> but depending on whether we are creating a new slot or reusing an existing one,
123        // we need to subtract the size of a slot (for creating a new slot) or 0 (for reusing an existing slot)
124        let new_free_space = header_mut.get_free_space()?
125            - bytes.len() as u16
126            - if plan.inserting_new_slot {
127                SLOT_SIZE as u16
128            } else {
129                0
130            };
131        header_mut.set_free_space(new_free_space)?;
132
133        // Insert the actual row in the page
134        self.data[start_offset..(start_offset + bytes.len())].copy_from_slice(bytes.as_slice());
135
136        // Right now, the header should contain the latest information regarding the number of slots (whether we added a new one or not),
137        // so the slot array should include any new bytes segment. We place the new slot directly at the requested index
138        self.slot_array_mut()?.set_slot(
139            plan.slot_number as u32,
140            start_offset as u16,
141            bytes.len() as u16,
142        );
143
144        Ok(())
145    }
