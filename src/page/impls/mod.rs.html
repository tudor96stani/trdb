1//! Module defining a fixed-size slotted page structure with its associated methods.
2//!
3//! # Memory Layout Overview
4//!
5//! A typical slotted page has this physical structure (generalized):
6//!
7//! ```text
8//!   ┌───────────────────────────────────────────────────────────────┐
9//!   │ Page Header (contains slot_count, free space ptrs, etc.)      │
10//!   ├───────────────────────────────────────────────────────────────┤
11//!   │ Tuple Data Region (grows upward)                              │
12//!   │   records / row fragments                                     │
13//!   │   variable sized                                              │
14//!   │   aligned upwards                                             │
15//!   ├───────────────────────────────────────────────────────────────┤
16//!   │ Free Space                                                    │
17//!   ├───────────────────────────────────────────────────────────────┤
18//!   │ Slot Array Region (grows downward)                            │
19//!   │   fixed-size SLOT_SIZE entries                                │
20//!   │   indexed logically left-to-right,                            │
21//!   │   stored physically right-to-left                             │
22//!   └───────────────────────────────────────────────────────────────┘
23//!
24//!                     ↑ page_start                        page_end ↑
25//! ```
26//!
27//! # Why This Design?
28//!
29//! - Adding a new slot does **not** require moving existing slots.
30//! - Tuple movement and compaction only affect the data region.
31//! - Both read and write operations are zero-copy and O(1).
32//!
33//! This module encapsulates that logic cleanly, exposing a safe and API for manipulating the slotted page.
34//!
35//!
36//! Header access is provided via `header::HeaderRef` and `header::HeaderMut` types.
37//! Slot array access is provided via `slot::SlotArrayRef` and `slot::SlotArrayMut` types.

59/// Wrapper around a fixed-size byte array representing a page.
60#[derive(Debug)]
61pub struct Page {
62    /// Unique identifier of the page. Comprised of file_name_hash::page_number_within_file
63    page_id: PageId,
64    /// Main binary array holding the `PAGE_SIZE` bytes of data for the page. Boxed and owned by this struct.
65    data: Box<[u8; PAGE_SIZE]>,
66}
67
68/// Public APIs for the Page struct.
69/// All public APIs use the `PageResult` type
70impl Page {
71    /// Retrieves a row from the page by its slot index.
72    ///
73    /// # Arguments
74    ///
75    /// * `slot_index` - The index of the slot to retrieve the row from. Indexing starts from 0.
76    ///
77    /// # Returns
78    ///
79    /// * `PageResult<&[u8]>` - A result containing a reference to the row data as a byte slice
80    ///   if successful, or an error wrapped in `PageResult` if the operation fails.
81    ///
82    /// # Errors
83    ///
84    /// This method can return the following errors:
85    /// * `PageError` - If there is an issue with the operation, such as an invalid slot index.
86    ///
87    /// The error is augmented with the `page_id` of the current page for better traceability.
88    pub fn row(&self, slot_index: u32) -> PageResult<&[u8]> {
89        self.read_row_internal(slot_index)
90            .map_err(PageOpError::from)
91            .with_page_id(self.page_id)
92    }
93
94    /// Plans the insertion of a row into the page. Used only for heap pages.
95    ///
96    /// # Arguments
97    ///
98    /// * `row_len` - The length of the row to be inserted, in bytes.
99    ///
100    /// # Returns
101    ///
102    /// * `PageResult<InsertionPlan>` - A result containing the insertion plan if successful,
103    ///   or an error wrapped in `PageResult` if the operation fails.
104    ///
105    /// # Errors
106    ///
107    /// This method can return the following errors:
108    /// * `PageError` - If there is an issue with the operation, such as insufficient space
109    ///   or other constraints preventing the insertion.
110    ///
111    /// The error is augmented with the `page_id` of the current page for better traceability.
112    pub fn plan_insert(&self, row_len: usize) -> PageResult<InsertionPlan> {
113        self.plan_insert_internal(row_len)
114            .map_err(PageOpError::from)
115            .with_page_id(self.page_id)
116    }
117
118    /// Inserts a row into a heap page using the provided insertion plan.
119    ///
120    /// # Arguments
121    ///
122    /// * `plan` - The `InsertionPlan` that specifies where and how the row should be inserted.
123    /// * `row` - A `Vec<u8>` containing the row data to be inserted.
124    ///
125    /// # Returns
126    ///
127    /// * `PageResult<()>` - A result indicating success (`Ok(())`) or failure (`Err(PageError)`).
128    ///
129    /// # Errors
130    ///
131    /// This method can return the following errors:
132    /// * `PageOpError` - If there is an issue during the insertion process. `PageOpError` will contain the source error.
133    /// * The error is augmented with the `page_id` of the current page for better traceability.
134    pub fn insert_heap(&mut self, plan: InsertionPlan, row: Vec<u8>) -> PageResult<()> {
135        self.insert_row_unsorted_internal(plan, row)
136            .map_err(PageOpError::from)
137            .with_page_id(self.page_id)
138    }
139}
