1//! Module defining a fixed-size slotted page structure with its associated methods.
2//!
3//! # Memory Layout Overview
4//!
5//! A typical slotted page has this physical structure (generalized):
6//!
7//! ```text
8//!   ┌───────────────────────────────────────────────────────────────┐
9//!   │ Page Header (contains slot_count, free space ptrs, etc.)      │
10//!   ├───────────────────────────────────────────────────────────────┤
11//!   │ Tuple Data Region (grows upward)                              │
12//!   │   records / row fragments                                     │
13//!   │   variable sized                                              │
14//!   │   aligned upwards                                             │
15//!   ├───────────────────────────────────────────────────────────────┤
16//!   │ Free Space                                                    │
17//!   ├───────────────────────────────────────────────────────────────┤
18//!   │ Slot Array Region (grows downward)                            │
19//!   │   fixed-size SLOT_SIZE entries                                │
20//!   │   indexed logically left-to-right,                            │
21//!   │   stored physically right-to-left                             │
22//!   └───────────────────────────────────────────────────────────────┘
23//!
24//!                     ↑ page_start                        page_end ↑
25//! ```
26//!
27//! # Why This Design?
28//!
29//! - Adding a new slot does **not** require moving existing slots.
30//! - Tuple movement and compaction only affect the data region.
31//! - Both read and write operations are zero-copy and O(1).
32//!
33//! This module encapsulates that logic cleanly, exposing a safe and API for manipulating the slotted page.
34//!
35//!
36//! Header access is provided via `header::HeaderRef` and `header::HeaderMut` types.
37//! Slot array access is provided via `slot::SlotArrayRef` and `slot::SlotArrayMut` types.

54/// Wrapper around a fixed-size byte array representing a page.
55#[derive(Debug)]
56pub struct Page {
57    /// Unique identifier of the page. Comprised of file_name_hash::page_number_within_file
58    page_id: PageId,
59    /// Main binary array holding the `PAGE_SIZE` bytes of data for the page. Boxed and owned by this struct.
60    data: Box<[u8; PAGE_SIZE]>,
61}
62
63/// Public APIs for the Page struct.
64/// All public APIs use the `PageResult` type
65impl Page {
66    /// Retrieves a row from the page by its slot index.
67    ///
68    /// # Arguments
69    ///
70    /// * `slot_index` - The index of the slot to retrieve the row from. Indexing starts from 0.
71    ///
72    /// # Returns
73    ///
74    /// * `PageResult<&[u8]>` - A result containing a reference to the row data as a byte slice
75    ///   if successful, or an error wrapped in `PageResult` if the operation fails.
76    ///
77    /// # Errors
78    ///
79    /// This method can return the following errors:
80    /// * `PageError` - If there is an issue with the operation, such as an invalid slot index.
81    ///
82    /// The error is augmented with the `page_id` of the current page for better traceability.
83    pub fn row(&self, slot_index: u32) -> PageResult<&[u8]> {
84        self.read_row_internal(slot_index)
85            .map_err(PageOpError::from)
86            .with_page_id(self.page_id)
87    }
88
89    /// Plans the insertion of a row into the page. Used only for heap pages.
90    ///
91    /// # Arguments
92    ///
93    /// * `row_len` - The length of the row to be inserted, in bytes.
94    ///
95    /// # Returns
96    ///
97    /// * `PageResult<InsertionPlan>` - A result containing the insertion plan if successful,
98    ///   or an error wrapped in `PageResult` if the operation fails.
99    ///
100    /// # Errors
101    ///
102    /// This method can return the following errors:
103    /// * `PageError` - If there is an issue with the operation, such as insufficient space
104    ///   or other constraints preventing the insertion.
105    ///
106    /// The error is augmented with the `page_id` of the current page for better traceability.
107    pub fn plan_insert(&self, row_len: usize) -> PageResult<InsertionPlan> {
108        self.plan_insert_internal(row_len)
109            .map_err(PageOpError::from)
110            .with_page_id(self.page_id)
111    }
112
113    /// Inserts a row into a heap page using the provided insertion plan.
114    ///
115    /// # Arguments
116    ///
117    /// * `plan` - The `InsertionPlan` that specifies where and how the row should be inserted.
118    /// * `row` - A `Vec<u8>` containing the row data to be inserted.
119    ///
120    /// # Returns
121    ///
122    /// * `PageResult<()>` - A result indicating success (`Ok(())`) or failure (`Err(PageError)`).
123    ///
124    /// # Errors
125    ///
126    /// This method can return the following errors:
127    /// * `PageOpError` - If there is an issue during the insertion process. `PageOpError` will contain the source error.
128    /// * The error is augmented with the `page_id` of the current page for better traceability.
129    pub fn insert_heap(&mut self, plan: InsertionPlan, row: Vec<u8>) -> PageResult<()> {
130        self.insert_row_unsorted_internal(plan, row)
131            .map_err(PageOpError::from)
132            .with_page_id(self.page_id)
133    }
134
135    /// Deletes a row from the page at the specified slot index.
136    ///
137    /// # Arguments
138    ///
139    /// * `slot_index` - The index of the slot from which the row should be deleted.
140    ///   Indexing starts from 0.
141    /// * `compact_requested` - A boolean flag indicating whether compaction should
142    ///   be performed after the deletion. If `true`, the page may attempt to compact
143    ///   the remaining rows to reclaim space.
144    ///
145    /// # Returns
146    ///
147    /// * `PageResult<()>` - A result indicating success (`Ok(())`) or failure
148    ///   (`Err(PageOpError)`). The error is augmented with the `page_id` of the
149    ///   current page for better traceability.
150    ///
151    /// # Errors
152    ///
153    /// This method can return the following errors:
154    /// * `PageOpError` - If there is an issue during the deletion process, such as
155    ///   an invalid slot index or other constraints preventing the deletion.
156    pub fn delete_row(&mut self, slot_index: usize, compact_requested: bool) -> PageResult<()> {
157        self.delete_row_internal(slot_index, compact_requested)
158            .map_err(PageOpError::from)
159            .with_page_id(self.page_id)
160    }
161
162    /// Updates the contents of a row.
163    /// New content can be of same size, smaller or larger.
164    /// For larger content, the new row size must still fit within the current page. It there is not enough room, an error will be returned to indicate this.
165    /// For such scenarios, a deletion & reinsertion is necessary.
166    ///
167    /// # Arguments
168    ///
169    /// * `slot_index`: the slot number of the row being updated
170    /// * `row`: the new content of the row.
171    ///
172    /// # Returns
173    ///
174    /// * `PageResult<()>` - A result indicating success (`Ok(())`) or failure
175    ///   (`Err(PageOpError)`). The error is augmented with the `page_id` of the
176    ///   current page for better traceability.
177    ///
178    /// # Errors
179    ///
180    /// This method can return the following errors:
181    /// * `PageOpError` - If there is an issue during the update process, such as
182    ///   an invalid slot index or other constraints preventing the update.
183    pub fn update_row(&mut self, slot_index: usize, row: Vec<u8>) -> PageResult<()> {
184        self.update_internal(slot_index, row)
185            .map_err(PageOpError::from)
186            .with_page_id(self.page_id)
187    }
188}
