9// TODO move this to public API
10/// Methods for creating and initializing pages.
11impl Page {
12    /// Creates a new page with all bytes initialized to zero. Private constructor.
13    pub fn new_zeroed(page_id: PageId) -> Self {
14        Self {
15            page_id,
16            data: Box::new([0; PAGE_SIZE]),
17        }
18    }
19
20    /// Creates a new page from an existing byte array.
21    pub fn new_from_bytes(bytes: Box<[u8; 4096]>, page_id: PageId) -> Self {
22        Self {
23            data: bytes,
24            page_id,
25        }
26    }
27
28    /// Creates a new empty page with the specified page ID and page type.
29    pub(crate) fn new_empty(page_id: PageId, page_type: PageType) -> Result<Self, HeaderError> {
30        let mut page = Self::new_zeroed(page_id);
31
32        page.header_mut()?.default(page_id.page_number, page_type);
33
34        Ok(page)
35    }
36
37    /// Initializes a page for the given `PageId` and `PageType`
38    /// Beware, this method will wipe out the contents of the internal byte array, zero-ing them out.
39    pub fn initialize(&mut self, page_id: PageId, page_type: PageType) -> PageResult<()> {
40        // Completely wipe the page by zero-ing it out.
41        (&mut *self.data)[..].fill(0);
42
43        let mut header = self
44            .header_mut()
45            .map_err(PageOpError::from)
46            .with_page_id(page_id)?;
47
48        // And reset the header for a fresh page.
49        header
50            .default(page_id.page_number, page_type)
51            .map_err(PageOpError::from)
52            .with_page_id(page_id)?;
53
54        Ok(())
55    }
56}

64    fn test_new_empty_page() {
65        let page_id = PageId::new(1, 0);
66        let page = Page::new_empty(page_id, PageType::Unsorted).unwrap();
67
68        assert_eq!(page.page_id(), page_id);
69
70        let header = page.header_ref().unwrap();
71        assert_eq!(header.get_page_number().unwrap(), 0);
72        assert_eq!(
73            header.get_page_type().unwrap(),
74            u16::from(PageType::Unsorted)
75        );
76    }
