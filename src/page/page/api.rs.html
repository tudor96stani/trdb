1use crate::PAGE_SIZE;
2use crate::errors::page_error::{PageResult, WithPageId};
3use crate::errors::page_op_error::PageOpError;
4use crate::insertion_plan::InsertionPlan;
5use crate::page_id::PageId;

8/// Wrapper around a fixed-size byte array representing a page.
9#[derive(Debug)]
10pub struct Page {
11    /// Unique identifier of the page. Comprised of file_name_hash::page_number_within_file
12    pub(crate) page_id: PageId,
13    /// Main binary array holding the `PAGE_SIZE` bytes of data for the page. Boxed and owned by this struct.
14    pub(crate) data: Box<[u8; PAGE_SIZE]>,
15}
16
17/// Public APIs for the Page struct.
18/// All public APIs use the `PageResult` type
19impl Page {
20    /// Creates a new page with all bytes initialized to zero. Private constructor.
21    pub fn new_zeroed(page_id: PageId) -> Self {
22        Self {
23            page_id,
24            data: Box::new([0; PAGE_SIZE]),
25        }
26    }
27
28    /// Initializes a page for the given `PageId` and `PageType`
29    /// Beware, this method will wipe out the contents of the internal byte array, zero-ing them out.
30    pub fn initialize(&mut self, page_id: PageId, page_type: PageType) -> PageResult<()> {
31        // Completely wipe the page by zero-ing it out.
32        (&mut *self.data)[..].fill(0);
33
34        let mut header = self
35            .header_mut()
36            .map_err(PageOpError::from)
37            .with_page_id(page_id)?;
38
39        // And reset the header for a fresh page.
40        header
41            .default(page_id.page_number, page_type)
42            .map_err(PageOpError::from)
43            .with_page_id(page_id)?;
44
45        Ok(())
46    }
47
48    /// Retrieves a row from the page by its slot index.
49    ///
50    /// # Arguments
51    ///
52    /// * `slot_index` - The index of the slot to retrieve the row from. Indexing starts from 0.
53    ///
54    /// # Returns
55    ///
56    /// * `PageResult<&[u8]>` - A result containing a reference to the row data as a byte slice
57    ///   if successful, or an error wrapped in `PageResult` if the operation fails.
58    ///
59    /// # Errors
60    ///
61    /// This method can return the following errors:
62    /// * `PageError` - If there is an issue with the operation, such as an invalid slot index.
63    ///
64    /// The error is augmented with the `page_id` of the current page for better traceability.
65    pub fn row(&self, slot_index: u32) -> PageResult<&[u8]> {
66        self.read_row_internal(slot_index)
67            .map_err(PageOpError::from)
68            .with_page_id(self.page_id)
69    }
70
71    /// Plans the insertion of a row into the page. Used only for heap pages.
72    ///
73    /// # Arguments
74    ///
75    /// * `row_len` - The length of the row to be inserted, in bytes.
76    ///
77    /// # Returns
78    ///
79    /// * `PageResult<InsertionPlan>` - A result containing the insertion plan if successful,
80    ///   or an error wrapped in `PageResult` if the operation fails.
81    ///
82    /// # Errors
83    ///
84    /// This method can return the following errors:
85    /// * `PageError` - If there is an issue with the operation, such as insufficient space
86    ///   or other constraints preventing the insertion.
87    ///
88    /// The error is augmented with the `page_id` of the current page for better traceability.
89    pub fn plan_insert(&self, row_len: usize) -> PageResult<InsertionPlan> {
90        self.plan_insert_internal(row_len)
91            .map_err(PageOpError::from)
92            .with_page_id(self.page_id)
93    }
94
95    /// Inserts a row into a heap page using the provided insertion plan.
96    ///
97    /// # Arguments
98    ///
99    /// * `plan` - The `InsertionPlan` that specifies where and how the row should be inserted.
100    /// * `row` - A `Vec<u8>` containing the row data to be inserted.
101    ///
102    /// # Returns
103    ///
104    /// * `PageResult<()>` - A result indicating success (`Ok(())`) or failure (`Err(PageError)`).
105    ///
106    /// # Errors
107    ///
108    /// This method can return the following errors:
109    /// * `PageOpError` - If there is an issue during the insertion process. `PageOpError` will contain the source error.
110    /// * The error is augmented with the `page_id` of the current page for better traceability.
111    pub fn insert_heap(&mut self, plan: InsertionPlan, row: Vec<u8>) -> PageResult<()> {
112        self.insert_row_unsorted_internal(plan, row)
113            .map_err(PageOpError::from)
114            .with_page_id(self.page_id)
115    }
116
117    /// Deletes a row from the page at the specified slot index.
118    ///
119    /// # Arguments
120    ///
121    /// * `slot_index` - The index of the slot from which the row should be deleted.
122    ///   Indexing starts from 0.
123    /// * `compact_requested` - A boolean flag indicating whether compaction should
124    ///   be performed after the deletion. If `true`, the page may attempt to compact
125    ///   the remaining rows to reclaim space.
126    ///
127    /// # Returns
128    ///
129    /// * `PageResult<()>` - A result indicating success (`Ok(())`) or failure
130    ///   (`Err(PageOpError)`). The error is augmented with the `page_id` of the
131    ///   current page for better traceability.
132    ///
133    /// # Errors
134    ///
135    /// This method can return the following errors:
136    /// * `PageOpError` - If there is an issue during the deletion process, such as
137    ///   an invalid slot index or other constraints preventing the deletion.
138    pub fn delete_row(&mut self, slot_index: usize, compact_requested: bool) -> PageResult<()> {
139        self.delete_row_internal(slot_index, compact_requested)
140            .map_err(PageOpError::from)
141            .with_page_id(self.page_id)
142    }
143
144    /// Updates the contents of a row.
145    /// New content can be of same size, smaller or larger.
146    /// For larger content, the new row size must still fit within the current page. It there is not enough room, an error will be returned to indicate this.
147    /// For such scenarios, a deletion & reinsertion is necessary.
148    ///
149    /// # Arguments
150    ///
151    /// * `slot_index`: the slot number of the row being updated
152    /// * `row`: the new content of the row.
153    ///
154    /// # Returns
155    ///
156    /// * `PageResult<()>` - A result indicating success (`Ok(())`) or failure
157    ///   (`Err(PageOpError)`). The error is augmented with the `page_id` of the
158    ///   current page for better traceability.
159    ///
160    /// # Errors
161    ///
162    /// This method can return the following errors:
163    /// * `PageOpError` - If there is an issue during the update process, such as
164    ///   an invalid slot index or other constraints preventing the update.
165    pub fn update_row(&mut self, slot_index: usize, row: Vec<u8>) -> PageResult<()> {
166        self.update_internal(slot_index, row)
167            .map_err(PageOpError::from)
168            .with_page_id(self.page_id)
169    }
170
171    /// Returns an immutable reference to the underlying data of the page.
172    pub fn data(&self) -> &[u8; PAGE_SIZE] {
173        &self.data
174    }
175
176    /// Returns a mutable reference to the underlying byte array of the page
177    pub fn data_mut(&mut self) -> &mut [u8; PAGE_SIZE] {
178        &mut self.data
179    }
180}
