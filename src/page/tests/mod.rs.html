12/// This section defines helper methods for the test suite. They are defined as methods on the `Page` struct, but are only available in the test environment.
13#[cfg(test)]
14impl Page {
15    /// Returns a new empty heap page for testing purposes
16    pub(super) fn test_create_empty_heap() -> Page {
17        Page::new_empty(PageId::new(1, 1), PageType::Unsorted).unwrap()
18    }
19
20    /// Sets the provided slot count in the header and updates the free_end field accordingly.
21    pub(super) fn test_set_slot_count(&mut self, count: usize) {
22        // 4096 - count*slot_size - 1
23        let mut header_mut = self.header_mut().unwrap();
24        header_mut.set_slot_count(count as u16).unwrap();
25
26        let new_free_end = PAGE_SIZE - count * SLOT_SIZE - 1;
27        header_mut.set_free_end(new_free_end as u16);
28    }
29
30    /// Populates the slot array with the provided slots. Sets up the header with the correct slot count as well.
31    pub(super) fn test_create_slots(&mut self, slots: Vec<SlotValues>) {
32        self.test_set_slot_count(slots.len());
33        let mut slot_array_mut = self.slot_array_mut().unwrap();
34        for (index, slot) in slots.iter().enumerate() {
35            slot_array_mut
36                .slot_mut(index as u32)
37                .unwrap()
38                .set_offset(slot.offset as u16)
39                .unwrap();
40            slot_array_mut
41                .slot_mut(index as u32)
42                .unwrap()
43                .set_length(slot.len as u16)
44                .unwrap();
45        }
46    }
47
48    /// Inserts a set of rows in the page, as instructed by the slot array provided. Each row will have its `index_in_slot_array + 1` as binary value repeated for the `len` of the slot.
49    /// ### Example
50    /// ```rust
51    /// let slots = vec![
52    //         SlotValues { offset: 100, len: 10 },
53    //         SlotValues { offset: 110, len: 20 },
54    //     ];
55    /// ```
56    /// will provide the following arrays:
57    /// ```rust
58    /// [1u8; 10] // for the 1st slot
59    /// [2u8; 20] // for the 2nd slot
60    /// ```
61    pub(super) fn test_insert_rows(&mut self, slots: Vec<SlotValues>) {
62        self.test_set_slot_count(slots.len());
63        for (index, slot) in slots.iter().enumerate() {
64            let mut slot_array_mut = self.slot_array_mut().unwrap();
65            slot_array_mut
66                .slot_mut(index as u32)
67                .unwrap()
68                .set_offset(slot.offset as u16)
69                .unwrap();
70            slot_array_mut
71                .slot_mut(index as u32)
72                .unwrap()
73                .set_length(slot.len as u16)
74                .unwrap();
75            let value = (index + 1) as u8;
76            let len = slot.len;
77            self.data_mut()[slot.offset..slot.offset + slot.len]
78                .copy_from_slice(vec![value; len].as_slice())
79        }
80    }
81
82    /// Asserts that at the provided `offset`, the `value` byte is repeated for `length` - basically that the row data is what is expected, circumventing going through the slot array.
83    pub(super) fn assert_row(&mut self, offset: usize, length: usize, value: u8) {
84        let mut actual_row_data = &mut self.data_mut()[offset..offset + length];
85
86        assert_eq!(vec![value; length].as_slice(), actual_row_data);
87    }
88
89    pub(super) fn assert_slot(&mut self, slot_index: usize, offset: usize, length: usize) {
90        let slot = self
91            .slot_array_ref()
92            .unwrap()
93            .slot_ref(slot_index as u32)
94            .unwrap();
95        assert_eq!(slot.offset().unwrap(), offset as u16);
96        assert_eq!(slot.length().unwrap(), length as u16);
97    }
98}

107/// This part generates helper methods on the error types, to more easily break them apart during assertions and get to the inner most error.

118    // Macro to generate expect_* helpers
119    macro_rules! impl_expect_ref {
120        ($enum:ty, $fn_name:ident, $variant:ident => $inner:ty) => {
121            impl $enum {
122                #[track_caller]
123                pub fn $fn_name(&self) -> &$inner {
124                    match self {
125                        Self::$variant(inner) => inner,
126                        other => panic!(
127                            "expected {}::{}, got {other:?}",
128                            stringify!($enum),
129                            stringify!($variant),
130                        ),
131                    }
132                }
133            }
134        };
135    }
136

138    impl_expect_ref!(PageOpError, expect_slot_error, Slot => SlotError);
139    impl_expect_ref!(PageOpError, expect_read_row_error, ReadRow => ReadRowError);
140    impl_expect_ref!(PageOpError, expect_insert_error, Insert => InsertError);
141    impl_expect_ref!(PageOpError, expect_delete_error, DeleteRow => DeleteError);
