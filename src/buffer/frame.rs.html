1use page::impls::Page;
2use page::page_id::PageId;
3use std::sync::RwLock;
4use std::sync::atomic::{AtomicBool, AtomicU32};
5
6/// The ID of a frame is basically just its index in the buffer's vector
7pub(crate) type FrameId = usize;
8
9/// A buffer frame is a memory allocation designed to store the contents of a data page in memory,
10/// along with other metadata needed by the buffer manager.
11/// The `page` is protected by a `RwLock` for concurrent access.
12/// The `page_id` is also protected by a `RwLock`. Additionally, the `page_id` is optional (`Option<T>`),
13/// to indicate that the frame is empty. An empty frame will contain a zeroed `Page`.
14///
15/// Access to the `BufferFrame` is not allowed outside the `BufferManager` - instead, `guard-like`
16/// structs will be used to provide references to the underlying data.
17///
18/// The `BufferFrame` is the owner of the `Page` - it consumes it during creation.
19/// The `Page` is never moved outside of the frame, only borrowed (either mutably or immutably)
20#[derive(Debug)]
21pub(crate) struct BufferFrame {
22    /// The `PageId` corresponding to the `Page` stored in the `page` field.
23    /// Optional. If frame is empty, this will be `None`
24    /// Protected by a `RwLock`
25    pub(crate) page_id: RwLock<Option<PageId>>,
26
27    /// The actual `Page` instance.
28    /// Protected by a `RwLock`
29    pub(crate) page: RwLock<Page>,
30
31    /// Atomic pin count for the `page`. Only to be used internally by the buffer manager's eviction policy.
32    pub(crate) pin_count: AtomicU32,
33
34    /// Dirtiness of the page flag, backed by an `AtomicBool`
35    pub(crate) dirty: AtomicBool,
36}
37
38impl Default for BufferFrame {
39    fn default() -> Self {
40        Self {
41            page_id: RwLock::new(None),
42            page: RwLock::new(Page::new_zeroed(PageId::new(0, 0))),
43            pin_count: AtomicU32::new(0),
44            dirty: AtomicBool::new(false),
45        }
46    }
47}
