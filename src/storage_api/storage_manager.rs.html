1//! The storage manager
2use buffer::buffer::BufferManager;
3use buffer::guards::{PageReadGuard, PageWriteGuard};
4use file::api::FileManager;
5use file::file_catalog::FileCatalog;
6use file::in_memory_file_manager::InMemoryFileManager;
7use page::impls::Page;
8use page::page_id::PageId;
9use std::sync::Arc;
10
11/// The storage manager
12#[derive(Debug)]
13pub struct StorageManager<F: FileManager> {
14    file_manager: Arc<F>,
15    buffer_manager: Arc<BufferManager<F>>,
16}
17
18impl<F: FileManager> StorageManager<F> {
19    /// Creates a new instance of the `StorageManager`
20    pub fn new(file_manager: Arc<F>, buffer_manager: Arc<BufferManager<F>>) -> Self {
21        Self {
22            file_manager,
23            buffer_manager,
24        }
25    }
26
27    /// Obtain a `&Page` via `PageReadGuard` for the provided `PageId`
28    pub fn read_page(&self, page_id: PageId) -> PageReadGuard<'_> {
29        match self.buffer_manager.read_page(page_id) {
30            Ok(page_read_guard) => page_read_guard,
31            Err(_) => panic!("buffer manager poisoned"),
32        }
33    }
34
35    /// Obtain a `&mut Page` via `PageWriteGuard` for the provided `PageId`
36    pub fn write_page(&self, page_id: PageId) -> PageWriteGuard<'_> {
37        match self.buffer_manager.read_page_mut(page_id) {
38            Ok(page_read_guard) => page_read_guard,
39            Err(_) => panic!("buffer manager poisoned"),
40        }
41    }
42
43    /// Initialize a new `Page` on the buffer for the provided `PageId` and obtain a `&mut Page`
44    /// via a `PageWriteGuard`
45    pub fn new_page(&self, page_id: PageId) -> PageWriteGuard<'_> {
46        match self.buffer_manager.allocate_new_page(page_id) {
47            Ok(page_write_guard) => page_write_guard,
48            Err(_) => panic!("buffer manager poisoned"),
49        }
50    }
51}
