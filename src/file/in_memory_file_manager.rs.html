1//! Defines a temporary implementation for the `InMemoryFileManager`
2//!
3use crate::api::FileManager;
4use crate::file_catalog::FileCatalog;
5use page::page_id::PageId;
6use std::collections::HashMap;
7use std::path::PathBuf;
8use std::sync::{Arc, RwLock};
9
10/// A temporary in memory file manager
11#[derive(Debug)]
12pub struct InMemoryFileManager {
13    data_dir: PathBuf,
14    // For now unused in this implementation, as filenames are not yet needed.
15    file_catalog: Arc<FileCatalog>,
16    pages: RwLock<HashMap<PageId, Box<[u8]>>>,
17}
18
19impl FileManager for InMemoryFileManager {
20    fn new<P>(path: P, file_catalog: Arc<FileCatalog>) -> Self
21    where
22        P: Into<PathBuf>,
23    {
24        Self {
25            data_dir: path.into(),
26            file_catalog,
27            pages: RwLock::new(HashMap::new()),
28        }
29    }
30
31    fn read_page(&self, page_id: PageId, destination: &mut [u8]) -> bool {
32        if let Some(page) = self.pages.read().unwrap().get(&page_id) {
33            destination.copy_from_slice(page);
34            true
35        } else {
36            false
37        }
38    }
39
40    fn write_page(&self, page_id: PageId, page_data: &[u8]) {
41        let boxed = page_data.to_vec().into_boxed_slice();
42        self.pages.write().unwrap().insert(page_id, boxed);
43    }
44}
