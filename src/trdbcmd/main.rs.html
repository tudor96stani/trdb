1//! Client for the TRDB server
2
3use std::env;
4use tokio::io;
5use tokio::io::{AsyncBufReadExt, AsyncReadExt, AsyncWriteExt};
6use tokio::net::TcpStream;
7
8// This is just a hardcoded client used for testing the communication with the server
9// but this crate will basically contain the CLI client that will read queries from the user,
10// send them to the server and print back the result.
11#[tokio::main]
12async fn main() -> Result<(), Box<dyn std::error::Error>> {
13    // If a cmd line arg was provided, use that one
14    let value: u32 = if let Some(arg) = env::args().nth(1) {
15        arg.trim().parse()?
16    } else {
17        // Otherwise read it from stdin
18        let mut stdin = io::BufReader::new(io::stdin());
19        let mut input = String::new();
20        println!("Enter a number to send to the server:");
21        stdin.read_line(&mut input).await?;
22        input.trim().parse()?
23    };
24
25    // Connect to the server
26    let mut stream = TcpStream::connect("127.0.0.1:8080").await?;
27
28    // Send the u32
29    stream.write_all(&value.to_le_bytes()).await?;
30
31    // Read response - up to 1024 bytes
32    let mut buf = vec![0u8; 1024];
33    let n = stream.read(&mut buf).await?;
34    let result = &buf[..n];
35
36    println!("Received: {:?}", result);
37
38    Ok(())
39}
