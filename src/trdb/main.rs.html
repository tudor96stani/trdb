1//! # TRDB Server
2//!
3//! This is the main entry point for the **TRDB** database server.
4//!
5//! The server is composed of multiple internal components organized under
6//! the `/crates` directory of this workspace:
7//!
8//! - `/storage`: Core storage engine handling data persistence and retrieval.
9//!
10//! This binary is the main executable for the database engine.
11
12#![allow(unused)] // Silence compiler warnings about unused code until they are referenced in main binary. TODO: remove this
13
14use crate::engine_environment::EngineEnvironment;
15use page::page_id;
16use page::page_id::PageId;
17use page::page_type::PageType;
18
19mod engine_environment;
20
21fn main() {
22    let e = EngineEnvironment::new();
23
24    let page_id = PageId::new(1, 1);
25    let mut page = e.storage.new_page(page_id);
26    let Ok(()) = page.initialize(page_id, PageType::Unsorted) else {
27        panic!("cannot set page type")
28    };
29
30    let Ok(insert_plan) = page.plan_insert(100) else {
31        panic!("insert plan failed");
32    };
33
34    match page.insert_heap(insert_plan, vec![1u8; 100]) {
35        Ok(_) => {}
36        Err(e) => panic!("insert failed: {}", e),
37    }
38
39    let Ok(row) = page.row(0) else {
40        panic!("row 0 failed");
41    };
42
43    println!("{:?}", row);
44}
