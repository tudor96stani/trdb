34// Temporarily placed a lot of logic in here for creating the TCP server, handling client requests, delegating them to the engine for processing, etc
35// All of this will be stripped into separate crates/modules, but for now it will do.
36#[tokio::main]
37async fn main() {
38    let logging_guard = init_logging("./logs");
39
40    let e = Arc::new(EngineEnvironment::new());
41
42    // dummy page
43    let page_id = PageId::new(1, 1);
44    {
45        let mut page = e.storage.new_page(page_id);
46        let Ok(()) = page.initialize(page_id, PageType::Unsorted) else {
47            panic!("cannot set page type")
48        };
49    }
50
51    let semaphore = Arc::new(Semaphore::new(8));
52    let shutdown = CancellationToken::new();
53
54    // Spawn a task that waits for OS shutdown signals and triggers cancellation.
55    {
56        let shutdown = shutdown.clone();
57        tokio::spawn(async move {
58            wait_for_shutdown_signal().await;
59            tracing::info!("shutdown signal received; beginning graceful shutdown");
60            shutdown.cancel();
61        });
62    }
63
64    let listener = TcpListener::bind("127.0.0.1:8080").await.unwrap();
65    tracing::info!("listening on {:?}", listener.local_addr());
66
67    let mut connections = JoinSet::new();
68
69    loop {
70        tokio::select! {
71            _ = shutdown.cancelled() => {
72                // Stop accepting new connections.
73                tracing::info!("stop accepting new connections");
74                break;
75            }
76
77            res = listener.accept() => {
78                let (socket, addr) = match res {
79                    Ok(v) => v,
80                    Err(e) => {
81                        if shutdown.is_cancelled() { break; }
82                        tracing::warn!("accept failed: {e}");
83                        continue;
84                    }
85                };
86
87                let env_clone = e.clone();
88                let semaphore_clone = semaphore.clone();
89                let shutdown_clone = shutdown.clone();
90
91                connections.spawn(async move {
92                    tracing::info!("client connected: {addr}");
93                    handle_client(socket, env_clone, semaphore_clone, shutdown_clone).await;
94                    tracing::info!("client disconnected: {addr}");
95                });
96            }
97        }
98    }
99
100    tracing::info!("waiting for existing connections to finish");
101    while let Some(res) = connections.join_next().await {
102        if let Err(join_err) = res {
103            tracing::warn!("connection task ended with error: {join_err}");
104        }
105    }
106
107    drop(logging_guard);
108    tracing::info!("shutdown complete");
109}

111async fn wait_for_shutdown_signal() {
112    #[cfg(unix)]
113    {
114        use tokio::signal::unix::{SignalKind, signal};
115
116        let mut sigint = signal(SignalKind::interrupt()).expect("sigint handler");
117        let mut sigterm = signal(SignalKind::terminate()).expect("sigterm handler");
118
119        tokio::select! {
120            _ = sigint.recv() => {},
121            _ = sigterm.recv() => {},
122        }
123    }
124
125    #[cfg(not(unix))]
126    {
127        tokio::signal::ctrl_c().await.expect("ctrl_c handler");
128    }
129}

131async fn handle_client(
132    mut socket: TcpStream,
133    env: Arc<EngineEnvironment>,
134    semaphore: Arc<Semaphore>,
135    shutdown: CancellationToken,
136) {
137    tracing::info!("client connected on {:?}", socket.peer_addr());
138
139    // for now the client only sends a u32.
140    let mut buf = [0u8; 4];
141
142    let read_res = tokio::select! {
143        _ = shutdown.cancelled() => {
144            tracing::info!("shutdown: stop reading new requests");
145            return;
146        }
147        r = socket.read_exact(&mut buf) => r,
148    };
149
150    if read_res.is_err() {
151        tracing::error!(
152            "error while reading data from socket for client {:?}",
153            socket.peer_addr()
154        );
155        return;
156    }
157    let _value = u32::from_le_bytes(buf);
158
159    // Ask for a permit from the semaphore
160    let permit = tokio::select! {
161        _ = shutdown.cancelled() => {
162            tracing::info!("shutdown: refuse starting new query");
163            return;
164        }
165        p = semaphore.acquire() => p.unwrap(),
166    };
167    // Start processing the client request on a separate, blocking thread
168    // a small note here: we request the work on a separate thread and wait for it to deliver a final result - this is fine for testing right now,
169    // but it's not the end goal: normally, we would want the worker thread to deliver results as they are processed, so that we can start streaming
170    // the result rows back to the client even if the worker is not yet fully done - once we have here enough rows to fill a packet, we can start shipping it via the TCP socket.
171    let env_clone = env.clone();
172    let row = task::spawn_blocking(move || process_query(env_clone, _value))
173        .await
174        .unwrap(); // await here to yield this thread back to the executor, while the query is being processed.
175
176    // drop the semaphore permit - we are done with the worker thread, so others can use it while we stream the results back to the client via the TCP stream
177    drop(permit);
178
179    // Send the result back to the client. For now, we only get a Vec<u8> from the worker thread, for testing purposes only. Normally we would get a result set that would be streamed as it is being processed.
180    if socket.write_all(&row).await.is_err() {
181        tracing::error!("error while writing result")
182    }
183}

185fn process_query(e: Arc<EngineEnvironment>, number: u32) -> Vec<u8> {
186    // Read the page with the hardcoded ID
187    let mut page = e.storage.write_page(PageId::new(1, 1));
188
189    // Attempt an insert of a 100bytes row
190    let Ok(insert_plan) = page.plan_insert(100) else {
191        panic!("insert plan failed");
192    };
193
194    // Create a row composed of 100 bytes with the provided number and insert it
195    let byte_value = number as u8;
196    match page.insert_heap(insert_plan, vec![byte_value; 100]) {
197        Ok(_) => {}
198        Err(e) => panic!("insert failed: {}", e),
199    }
200
201    // Re-read the row to ensure it was inserted
202    let Ok(row) = page.row(number - 1) else {
203        panic!("row 0 failed");
204    };
205
206    // Return the raw row binary data
207    Vec::from(row)
208}

210/// Sets up the logging for the server
211pub fn init_logging(log_dir: &str) -> Result<WorkerGuard, Box<dyn Error + Send + Sync>> {
212    let file_appender = tracing_appender::rolling::daily(log_dir, "trdb.log");
213    let (file_writer, guard) = tracing_appender::non_blocking(file_appender);
214
215    let filter = EnvFilter::builder()
216        .with_default_directive(LevelFilter::INFO.into())
217        .from_env_lossy();
218
219    let console_layer = fmt::layer()
220        .with_writer(std::io::stdout)
221        .with_target(false)
222        .with_level(true)
223        .compact();
224
225    let file_layer = fmt::layer()
226        .with_writer(file_writer)
227        .json()
228        .with_current_span(true)
229        .with_span_list(true);
230
231    tracing_subscriber::registry()
232        .with(filter)
233        .with(console_layer)
234        .with(file_layer)
235        .init();
236
237    Ok(guard)
238}
