11/// Owner of the singleton-like instances that are needed for the entire lifetime of the server
12#[derive(Debug)]
13pub struct EngineEnvironment {
14    pub file_manager: Arc<DiskFileManager>,
15    pub buffer: Arc<BufferManager<DiskFileManager>>,
16    pub storage: Arc<StorageManager<DiskFileManager>>,
17    pub file_catalog: Arc<FileCatalog>,
18    pub engine_config: EngineConfig,
19}
20
21impl EngineEnvironment {
22    pub fn new(config: EngineConfig) -> Self {
23        let file_catalog = Arc::new(FileCatalog::new());
24        let file_manager = Arc::new(DiskFileManager::new(file_catalog.clone()));
25        let buffer = Arc::new(BufferManager::new(
26            file_manager.clone(),
27            config.storage.buffer_pages.get(),
28        ));
29        let storage = Arc::new(StorageManager::new(file_manager.clone(), buffer.clone()));
30        Self {
31            file_manager,
32            buffer,
33            storage,
34            file_catalog,
35            engine_config: config,
36        }
37    }
